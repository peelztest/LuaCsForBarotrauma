438 {
439 if (Math.Abs(attackResult.Damage) < 0.01f) { return; }
440 DamageOverlayTimer = MathHelper.Clamp(attackResult.Damage / MaxVitality, DamageOverlayTimer, 1.0f);
441 if (healthShadowDelay <= 0.0f) { healthShadowDelay = 1.0f; }
442
443 if (healthBarPulsateTimer <= 0.0f) { healthBarPulsatePhase = 0.0f; }
444 healthBarPulsateTimer = 1.0f;
445
446 float additionalIntensity = MathHelper.Lerp(0, 1, MathUtils.InverseLerp(0, 0.1f, attackResult.Damage / MaxVitality));
447 damageIntensity = MathHelper.Clamp(damageIntensity + additionalIntensity, 0, 1);
448
449 DisplayVitalityDelay = 0.5f;
450 }

547 partial void UpdateBleedingProjSpecific(AfflictionBleeding affliction, Limb targetLimb, float deltaTime)
548 {
549 if (Character.InvisibleTimer > 0.0f) { return; }
550
551 bloodParticleTimer -= deltaTime * (affliction.Strength / 10.0f);
552 if (bloodParticleTimer <= 0.0f)
553 {
554 Limb limb = targetLimb ?? Character.AnimController.MainLimb;
555
556 bool inWater = Character.AnimController.InWater;
557 var drawTarget = inWater ? Particles.ParticlePrefab.DrawTargetType.Water : Particles.ParticlePrefab.DrawTargetType.Air;
558 var emitter = Character.BloodEmitters.FirstOrDefault(e => e.Prefab.ParticlePrefab.DrawTarget == drawTarget || e.Prefab.ParticlePrefab.DrawTarget == Particles.ParticlePrefab.DrawTargetType.Both);
559 float particleMinScale = emitter?.Prefab.Properties.ScaleMin ?? 0.5f;
560 float particleMaxScale = emitter?.Prefab.Properties.ScaleMax ?? 1;
561 float severity = Math.Min(affliction.Strength / affliction.Prefab.MaxStrength * Character.Params.BleedParticleMultiplier, 1);
562 float bloodParticleSize = MathHelper.Lerp(particleMinScale, particleMaxScale, severity);
563
564 Vector2 velocity = Rand.Vector(affliction.Strength * 0.1f);
565 if (!inWater)
566 {
567 bloodParticleSize *= 2.0f;
568 velocity = limb.LinearVelocity * 100.0f;
569 }
570
571 // TODO: use the blood emitter?
572 var blood = GameMain.ParticleManager.CreateParticle(
573 inWater ? Character.Params.BleedParticleWater : Character.Params.BleedParticleAir,
574 limb.WorldPosition, velocity, 0.0f, Character.AnimController.CurrentHull);
575
576 if (blood != null)
577 {
578 blood.Size *= bloodParticleSize;
579 if (!inWater && !string.IsNullOrEmpty(Character.BloodDecalName) && Rand.Range(0.0f, 1.0f) < 0.05f)
580 {
581 blood.OnCollision += (Vector2 pos, Hull hull) =>
582 {
583 var decal = hull?.AddDecal(Character.BloodDecalName, pos, Rand.Range(1.0f, 2.0f), isNetworkEvent: true);
584 if (decal != null)
585 {
586 decal.FadeTimer = decal.LifeTime - decal.FadeOutTime * 2;
587 }
588 };
589 }
590 }
591 bloodParticleTimer = MathHelper.Lerp(2, 0.5f, severity);
592 }

619 {
620 forceAfflictionContainerUpdate = true;
621 currentDisplayedAfflictions = GetAllAfflictions(mergeSameAfflictions: true)
622 .FindAll(a => a.ShouldShowIcon(Character) && a.Prefab.Icon != null);
623 currentDisplayedAfflictions.Sort((a1, a2) =>
624 {
625 int dmgPerSecond = Math.Sign(a1.DamagePerSecond - a2.DamagePerSecond);
626 if (dmgPerSecond != 0) { return dmgPerSecond; }
627 return Math.Sign(GetStr(a1) - GetStr(a2));
628 static float GetStr(Affliction affliction)
629 {
630 return affliction.Strength / affliction.Prefab.MaxStrength * (affliction.Prefab.IsBuff ? 1.0f : 10.0f);
631 }

638 {

640 }

644 Math.Min(healthShadowSize + deltaTime, healthBar.BarSize) :
645 Math.Max(healthShadowSize - deltaTime, healthBar.BarSize);

648 float blurStrength = 0.0f;
649 float distortStrength = 0.0f;
650 float distortSpeed = 0.0f;
651 float radialDistortStrength = 0.0f;
652 float chromaticAberrationStrength = 0.0f;
653 float grainStrength = 0.0f;
654 Color grainColor = Color.Transparent;
655
656 float oxygenLowStrength = 0.0f;
657 if (Character.IsUnconscious)
658 {
659 blurStrength = 1.0f;
660 distortSpeed = 1.0f;
661 }
662 else if (OxygenAmount < 100.0f)
663 {
664 oxygenLowStrength = Math.Min(1.0f - (OxygenAmount - LowOxygenThreshold) / LowOxygenThreshold, 1.0f);
665 blurStrength = MathHelper.Lerp(0.5f, 1.0f, 1.0f - Vitality / MaxVitality) * oxygenLowStrength;
666 distortStrength = blurStrength * oxygenLowStrength;
667 distortSpeed = blurStrength + 1.0f;
668 distortSpeed *= distortSpeed * distortSpeed * distortSpeed;
669
670 grainStrength = MathHelper.Lerp(0.5f, 10.0f, oxygenLowStrength);
671 grainColor = oxygenLowGrainColor;

674 foreach (KeyValuePair<Affliction, LimbHealth> kvp in afflictions)
675 {
676 var affliction = kvp.Key;

1567 strength -= reduceAffliction.ReduceAmount * (effect.Duration > 0 ? effect.Duration : 1.0f);

1569 foreach (var addAffliction in effect.Afflictions)
1570 {
1571 if (addAffliction.Prefab != affliction.Prefab) { continue; }
1572 strength += addAffliction.Strength * (effect.Duration > 0 ? effect.Duration : 1.0f);
1573 }
1574 }
1575 }

1928 newPeriodicEffects.Add((afflictionPrefab.PeriodicEffects[j], periodicAfflictionTimer));
1929 }
1930 newAfflictions.Add((null, afflictionPrefab, afflictionStrength));
1931 }
