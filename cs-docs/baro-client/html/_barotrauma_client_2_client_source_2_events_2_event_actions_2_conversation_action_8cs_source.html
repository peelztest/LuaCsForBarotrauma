141  static void RecalculateLastMessage(GUIListBox conversationList, bool append)
142  {
143  if (conversationList.Content.Children.LastOrDefault() is GUILayoutGroup lastElement)
144  {
145  GUILayoutGroup textLayout = lastElement.GetChild<GUILayoutGroup>();
146 
147  if (textLayout != null)
148  {
149  if (lastElement.Rect.Size.Y < textLayout.Rect.Size.Y && !append)
150  {
151  lastElement.RectTransform.MinSize = textLayout.Rect.Size;
152  }
153 
154  int textHeight = textLayout.Children.Sum(c => c.Rect.Height);
155  textLayout.RectTransform.MaxSize = new Point(lastElement.RectTransform.MaxSize.X, textHeight);
156  textLayout.Recalculate();
157  }
158  int sumHeight = lastElement.Children.Sum(c => c.Rect.Height);
159  lastElement.RectTransform.MaxSize = new Point(lastElement.RectTransform.MaxSize.X, sumHeight);
160  lastElement.Recalculate();
161  conversationList.RecalculateChildren();
162 
163  if (!append || textLayout == null) { return; }
164 
165  foreach (GUIComponent child in textLayout.Children)
166  {
167  conversationList.UpdateScrollBarSize();
168  float wait = conversationList.BarSize < 1.0f ? 0.5f : 0.0f;
169 
170  if (child is GUITextBlock) { child.FadeIn(wait, 0.5f); }
171 
172  if (child is GUIButton btn)
173  {
174  btn.FadeIn(wait, 1.0f);
175  btn.TextBlock.FadeIn(wait, 0.5f);
176  }
177  }
178  }
