53 private static void CreateDialog(string text, Character speaker, IEnumerable<string> options, int[] closingOptions, string spriteIdentifier = null,
54 ConversationAction actionInstance = null, UInt16? actionId = null, bool fadeToBlack = false, DialogTypes dialogType = DialogTypes.Regular, bool continueConversation = false)
55 {
56 Debug.Assert(actionInstance == null || actionId == null);
57
58 if (GUI.InputBlockingMenuOpen)
59 {
60 if (actionId.HasValue) { SendIgnore(actionId.Value); }
61 return;

63
64 shouldFadeToBlack = fadeToBlack;
65
66 if (lastMessageBox != null && !lastMessageBox.Closed && GUIMessageBox.MessageBoxes.Contains(lastMessageBox))
67 {
68 if (actionId != null && lastMessageBox.UserData is Pair<string, ushort> userData)
69 {
70 if (userData.Second == actionId) { return; }
71 lastMessageBox.UserData = new Pair<string, ushort>("ConversationAction", actionId.Value);
72 }
73
74 GUIListBox conversationList = lastMessageBox.FindChild("conversationlist", true) as GUIListBox;
75 Debug.Assert(conversationList != null);
76
77 // gray out the last text block
78 if (conversationList.Content.Children.LastOrDefault() is GUILayoutGroup lastElement)
79 {
80 if (lastElement.FindChild("text", true) is GUITextBlock textLayout)
81 {
82 textLayout.OverrideTextColor(Color.DarkGray * 0.8f);
83 }
84 }
85
86 List<GUIButton> extraButtons = CreateConversation(conversationList, text, speaker, options, string.IsNullOrWhiteSpace(spriteIdentifier));
87 AssignActionsToButtons(extraButtons, lastMessageBox);
88 RecalculateLastMessage(conversationList, true);
89
90 conversationList.ScrollToEnd(0.5f);
91 lastMessageBox.SetBackgroundIcon(EventSet.GetEventSprite(spriteIdentifier));
92 return;
93 }

95 var (relative, min) = GetSizes(dialogType);

97 GUIMessageBox messageBox = new GUIMessageBox(string.Empty, string.Empty, Array.Empty<LocalizedString>(),
98 relativeSize: relative, minSize: min,
99 type: GUIMessageBox.Type.InGame, backgroundIcon: EventSet.GetEventSprite(spriteIdentifier))
100 {
101 UserData = "ConversationAction"

103
104 lastMessageBox = messageBox;

106 messageBox.InnerFrame.ClearChildren();

108 GUIStyle.Apply(messageBox.InnerFrame, "DialogBox");
109
110 if (actionInstance != null)
111 {
112 lastActiveAction = actionInstance;

141 static void RecalculateLastMessage(GUIListBox conversationList, bool append)
142 {
143 if (conversationList.Content.Children.LastOrDefault() is GUILayoutGroup lastElement)
144 {
145 GUILayoutGroup textLayout = lastElement.GetChild<GUILayoutGroup>();
146
147 if (textLayout != null)
148 {
149 if (lastElement.Rect.Size.Y < textLayout.Rect.Size.Y && !append)
150 {
151 lastElement.RectTransform.MinSize = textLayout.Rect.Size;
152 }
153
154 int textHeight = textLayout.Children.Sum(c => c.Rect.Height);
155 textLayout.RectTransform.MaxSize = new Point(lastElement.RectTransform.MaxSize.X, textHeight);
156 textLayout.Recalculate();
157 }
158 int sumHeight = lastElement.Children.Sum(c => c.Rect.Height);
159 lastElement.RectTransform.MaxSize = new Point(lastElement.RectTransform.MaxSize.X, sumHeight);
160 lastElement.Recalculate();
161 conversationList.RecalculateChildren();
162
163 if (!append || textLayout == null) { return; }
164
165 foreach (GUIComponent child in textLayout.Children)
166 {
167 conversationList.UpdateScrollBarSize();
168 float wait = conversationList.BarSize < 1.0f ? 0.5f : 0.0f;
169
170 if (child is GUITextBlock) { child.FadeIn(wait, 0.5f); }
171
172 if (child is GUIButton btn)
173 {
174 btn.FadeIn(wait, 1.0f);
175 btn.TextBlock.FadeIn(wait, 0.5f);
176 }
177 }
178 }

181 void AssignActionsToButtons(List<GUIButton> optionButtons, GUIMessageBox target)
182 {
183 if (!options.Any())
184 {
185 GUIButton closeButton = new GUIButton(new RectTransform(Vector2.One, target.InnerFrame.RectTransform, Anchor.BottomRight, scaleBasis: ScaleBasis.Smallest)
186 {
187 MaxSize = new Point(GUI.IntScale(24)),
188 MinSize = new Point(24),
189 AbsoluteOffset = new Point(GUI.IntScale(48), GUI.IntScale(16))
190 }, style: "GUIButtonVerticalArrow")
191 {
192 UserData = "ContinueButton",
193 IgnoreLayoutGroups = true,
194 Bounce = true,
195 OnClicked = (btn, userdata) =>
196 {
197 if (actionInstance != null)
198 {
199 actionInstance.selectedOption = 0;
200 }
201 else if (actionId.HasValue)
202 {
203 SendResponse(actionId.Value, 0);
204 }
205
206 if (!continueConversation)
207 {
208 target.Close();
209 }
210 else
211 {
212 btn.Frame.FadeOut(0.33f, true);
213 }
214
215 return true;
216 }
217 };
218
219 double allowCloseTime = Timing.TotalTime + 0.5;
220 closeButton.Children.ForEach(child => child.SpriteEffects = SpriteEffects.FlipVertically);
221 closeButton.Frame.FadeIn(0.5f, 0.5f);
222 closeButton.SlideIn(0.5f, 0.33f, 16, SlideDirection.Down);
223
224 InputType? closeInput = null;
225 if (GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Use].MouseButton == MouseButton.None)
226 {
227 closeInput = InputType.Use;
228 }
229 else if (GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Select].MouseButton == MouseButton.None)
230 {
231 closeInput = InputType.Select;
232 }
233 if (closeInput.HasValue)
234 {
235 closeButton.ToolTip = TextManager.ParseInputTypes($"{TextManager.Get("Close")} ([InputType.{closeInput.Value}])");
236 closeButton.OnAddedToGUIUpdateList += (GUIComponent component) =>
237 {
238 if (Timing.TotalTime > allowCloseTime && PlayerInput.KeyHit(closeInput.Value))
239 {
240 GUIButton btn = component as GUIButton;
241 btn?.OnClicked(btn, btn.UserData);
242 btn?.Flash(GUIStyle.Green);
243 }
244 };
245 }
246 }
247
248 for (int i = 0; i < optionButtons.Count; i++)
249 {
250 optionButtons[i].UserData = i;
251 optionButtons[i].OnClicked += (btn, userdata) =>
252 {
253 int selectedOption = (userdata as int?) ?? 0;
254 if (actionInstance != null)
255 {
256 actionInstance.selectedOption = selectedOption;
257 foreach (GUIButton otherButton in optionButtons)
258 {
259 otherButton.CanBeFocused = false;
260 if (otherButton != btn)
261 {
262 otherButton.TextBlock.OverrideTextColor(Color.DarkGray * 0.8f);
263 }
264 }
265 btn.ExternalHighlight = true;
266 return true;
267 }
268
269 if (actionId.HasValue)
270 {
271 SendResponse(actionId.Value, selectedOption);
272 btn.CanBeFocused = false;
273 btn.ExternalHighlight = true;
274 foreach (GUIButton otherButton in optionButtons)
275 {
276 otherButton.CanBeFocused = false;
277 if (otherButton != btn)
278 {
279 otherButton.TextBlock.OverrideTextColor(Color.DarkGray * 0.8f);
280 }
281 }
282 return true;
283 }
284 //should not happen
285 return false;
286 };
287
288 if (closingOptions.Contains(i)) { optionButtons[i].OnClicked += target.Close; }
289 }
290 }

302 private static List<GUIButton> CreateConversation(GUIListBox parentBox, string text, Character speaker, IEnumerable<string> options, bool drawChathead = true)
303 {
304 var content = new GUILayoutGroup(new RectTransform(Vector2.One, parentBox.Content.RectTransform), childAnchor: Anchor.TopLeft, isHorizontal: true)
305 {
306 Stretch = true,
307 CanBeFocused = true,
308 AlwaysOverrideCursor = true
309 };
310
311 LocalizedString translatedText = TextManager.Get(text);
312
313 if (speaker?.Info != null && drawChathead)
314 {
315 // chathead
316 new GUICustomComponent(new RectTransform(new Vector2(0.15f, 0.8f), content.RectTransform), onDraw: (sb, customComponent) =>
317 {
318 speaker.Info.DrawIcon(sb, customComponent.Rect.Center.ToVector2(), customComponent.Rect.Size.ToVector2());
319 });
320 }
321
322 var textContent = new GUILayoutGroup(new RectTransform(new Vector2(1.0f, 0f), content.RectTransform), childAnchor: Anchor.TopCenter)
323 {
324 AbsoluteSpacing = GUI.IntScale(5)
325 };
326
327 var textBlock = new GUITextBlock(new RectTransform(new Vector2(1.0f, 0.0f), textContent.RectTransform), translatedText, wrap: true)
328 {
329 AlwaysOverrideCursor = true,
330 UserData = "text"
331 };
332
333 List<GUIButton> buttons = new List<GUIButton>();
334 if (options.Any())
335 {
336 foreach (string option in options)
337 {
338 var btn = new GUIButton(new RectTransform(new Vector2(0.9f, 0.01f), textContent.RectTransform), TextManager.Get(option), style: "ListBoxElement");
339 btn.TextBlock.TextAlignment = Alignment.CenterLeft;
340 btn.TextColor = btn.HoverTextColor = GUIStyle.Green;
341 btn.TextBlock.Wrap = true;
342 buttons.Add(btn);
343 }
344 }
345
346 content.Recalculate();
347 textContent.Recalculate();
348 textBlock.CalculateHeightFromText();
349 textBlock.RectTransform.MinSize = new Point(0, textBlock.Rect.Height);
350 foreach (GUIButton btn in buttons)
351 {
352 btn.TextBlock.SetTextPos();
353 btn.TextBlock.CalculateHeightFromText();
354 btn.RectTransform.MinSize = new Point(0, (int)(btn.TextBlock.Rect.Height * 1.2f));
355 }
356
357 textContent.RectTransform.MinSize = new Point(0, textContent.Children.Sum(c => c.Rect.Height) + GUI.IntScale(16));
358 content.RectTransform.MinSize = new Point(0, content.Children.Sum(c => c.Rect.Height));
359
360 // Recalculate the text size as it is scaled up and no longer matching the text height due to the textContent's minSize increasing
361 textBlock.CalculateHeightFromText();
362 textBlock.TextAlignment = Alignment.TopLeft;
363 //content.RectTransform.MinSize = new Point(0, textContent.Rect.Height);
364
365 return buttons;
366 }

368 private static void SendResponse(UInt16 actionId, int selectedOption)
369 {
370 IWriteMessage outmsg = new WriteOnlyMessage();
371 outmsg.Write((byte)ClientPacketHeader.EVENTMANAGER_RESPONSE);
372 outmsg.Write(actionId);
373 outmsg.Write((byte)selectedOption);
374 GameMain.Client?.ClientPeer?.Send(outmsg, DeliveryMethod.Reliable);
375 }
