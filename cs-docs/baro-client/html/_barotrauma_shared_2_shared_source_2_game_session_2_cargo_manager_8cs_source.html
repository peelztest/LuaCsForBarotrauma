296 public void PurchaseItems(Identifier storeIdentifier, List<PurchasedItem> itemsToPurchase, bool removeFromCrate, Client client = null)
297 {
298 var store = Location.GetStore(storeIdentifier);
299 if (store == null) { return; }
300 var itemsPurchasedFromStore = GetPurchasedItems(storeIdentifier, create: true);
301 // Check all the prices before starting the transaction to make sure the modifiers stay the same for the whole transaction
302 var buyValues = GetBuyValuesAtCurrentLocation(storeIdentifier, itemsToPurchase.Select(i => i.ItemPrefab));
303 var itemsInStoreCrate = GetBuyCrateItems(storeIdentifier, create: true);
304 foreach (PurchasedItem item in itemsToPurchase)
305 {
306 // Add to the purchased items
307 var purchasedItem = itemsPurchasedFromStore.Find(pi => pi.ItemPrefab == item.ItemPrefab);
308 if (purchasedItem != null)
309 {
310 purchasedItem.Quantity += item.Quantity;
311 }
312 else
313 {
314 purchasedItem = new PurchasedItem(item.ItemPrefab, item.Quantity, client);
315 itemsPurchasedFromStore.Add(purchasedItem);
316 }
317 // Exchange money
318 int itemValue = item.Quantity * buyValues[item.ItemPrefab];
319 campaign.TryPurchase(client, itemValue);
320 GameAnalyticsManager.AddMoneySpentEvent(itemValue, GameAnalyticsManager.MoneySink.Store, item.ItemPrefab.Identifier.Value);
321 store.Balance += itemValue;
322 if (removeFromCrate)
323 {
324 // Remove from the shopping crate
325 if (itemsInStoreCrate.Find(pi => pi.ItemPrefab == item.ItemPrefab) is { } crateItem)
326 {
327 crateItem.Quantity -= item.Quantity;
328 if (crateItem.Quantity < 1) { itemsInStoreCrate.Remove(crateItem); }
329 }
330 }
331 }
332 OnPurchasedItemsChanged?.Invoke();
333 }

425 private bool IsItemSellable(Item item, IEnumerable<SoldEntity> confirmedItems)
426 {
427 if (item.Removed) { return false; }
428 if (!item.Prefab.CanBeSold) { return false; }
429 if (item.SpawnedInCurrentOutpost) { return false; }
430 if (!item.Prefab.AllowSellingWhenBroken && item.ConditionPercentage < 90.0f) { return false; }
431 if (confirmedItems.Any(ci => ci.Item == item)) { return false; }
432 if (UndeterminedSoldEntities.TryGetValue(item.Prefab, out int count))
433 {
434 int newCount = count - 1;
435 if (newCount > 0)
436 {
437 UndeterminedSoldEntities[item.Prefab] = newCount;
438 }
439 else
440 {
441 UndeterminedSoldEntities.Remove(item.Prefab);
442 }
443 return false;
444 }
445 if (item.OwnInventory?.Container is ItemContainer itemContainer)
446 {
447 var containedItems = item.ContainedItems;
448 if (containedItems.None()) { return true; }
449 // Allow selling the item if contained items are unsellable and set to be removed on deconstruct
450 if (itemContainer.RemoveContainedItemsOnDeconstruct && containedItems.All(it => !it.Prefab.CanBeSold)) { return true; }
451 // Otherwise there must be no contained items or the contained items must be confirmed as sold
452 if (!containedItems.All(it => confirmedItems.Any(ci => ci.Item == it))) { return false; }
453 }
454 return true;
455 }

457 public static void CreateItems(List<PurchasedItem> itemsToSpawn, Submarine sub)
458 {
459 if (itemsToSpawn.Count == 0) { return; }
460
461 WayPoint wp = WayPoint.GetRandom(SpawnType.Cargo, null, sub);
462 if (wp == null)
463 {
464 DebugConsole.ThrowError("The submarine must have a waypoint marked as Cargo for bought items to be placed correctly!");
465 return;
466 }
467
468 Hull cargoRoom = Hull.FindHull(wp.WorldPosition);
469 if (cargoRoom == null)
470 {
471 DebugConsole.ThrowError("A waypoint marked as Cargo must be placed inside a room!");
472 return;
473 }
474
475 if (sub == Submarine.MainSub)
476 {
477#if CLIENT

479 TextManager.GetWithVariable("CargoSpawnNotification",

491 ChatMessageType.ServerMessageBoxInGame, null);

505 {

517 ep.Identifier == pi.ItemPrefab.CargoContainerIdentifier ||

524 }

527 Item containerItem = new Item(containerPrefab, containerPosition, wp.Submarine);
528 itemContainer = containerItem.GetComponent<ItemContainer>();

535#if SERVER

537 {

539 }

541 }

545 itemContainer?.Inventory.TryPutItem(item, null);

568 public static Vector2 GetCargoPos(Hull hull, ItemPrefab itemPrefab)
569 {
570 float floorPos = hull.Rect.Y - hull.Rect.Height;
571
572 Vector2 position = new Vector2(
573 hull.Rect.Width > 40 ? Rand.Range(hull.Rect.X + 20f, hull.Rect.Right - 20f) : hull.Rect.Center.X,
574 floorPos);
575
576 //check where the actual floor structure is in case the bottom of the hull extends below it
577 if (Submarine.PickBody(
578 ConvertUnits.ToSimUnits(new Vector2(position.X, hull.Rect.Y - hull.Rect.Height / 2)),
579 ConvertUnits.ToSimUnits(position),
580 collisionCategory: Physics.CollisionWall) != null)
581 {
582 float floorStructurePos = ConvertUnits.ToDisplayUnits(Submarine.LastPickedPosition.Y);
583 if (floorStructurePos > floorPos)
584 {
585 floorPos = floorStructurePos;
586 }
587 }
588
589 position.Y = floorPos + itemPrefab.Size.Y / 2;
590
591 return position;
592 }

612 public void LoadPurchasedItems(XElement element)
613 {
614 var purchasedItems = new Dictionary<Identifier, List<PurchasedItem>>();
615 if (element != null)
616 {
617 foreach (XElement itemElement in element.GetChildElements("item"))
618 {
619 string prefabId = itemElement.GetAttributeString("id", null);
620 if (string.IsNullOrWhiteSpace(prefabId)) { continue; }
621 var prefab = ItemPrefab.Prefabs.Find(p => p.Identifier == prefabId);
622 if (prefab == null) { continue; }
623 int qty = itemElement.GetAttributeInt("qty", 0);
624 Identifier storeId = itemElement.GetAttributeIdentifier("storeid", "merchant");
625 int buyerId = itemElement.GetAttributeInt("buyer", 0);
626 if (!purchasedItems.TryGetValue(storeId, out var storeItems))
627 {
628 storeItems = new List<PurchasedItem>();
629 purchasedItems.Add(storeId, storeItems);
630 }
631 storeItems.Add(new PurchasedItem(prefab, qty, buyerId));
632 }
633 }
634 SetPurchasedItems(purchasedItems);
635 }
