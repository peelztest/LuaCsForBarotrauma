581  private bool SelectTab(Tab tab)
582  {
583  titleText.Visible = true;
584  SettingsMenu.Instance?.Close();
585  #warning TODO: reimplement settings confirmation dialog
586 
587  switch (tab)
588  {
589  case Tab.NewGame:
590  if (GameSettings.CurrentConfig.TutorialSkipWarning)
591  {
592  selectedTab = Tab.Empty;
593  ShowTutorialSkipWarning(Tab.NewGame);
594  return true;
595  }
596  if (!GameSettings.CurrentConfig.CampaignDisclaimerShown)
597  {
598  selectedTab = Tab.Empty;
599  GameMain.Instance.ShowCampaignDisclaimer(() => { SelectTab(null, Tab.NewGame); });
600  return true;
601  }
602  campaignSetupUI.RandomizeCrew();
603  campaignSetupUI.SetPage(0);
604  campaignSetupUI.CreateDefaultSaveName();
605  campaignSetupUI.RandomizeSeed();
606  campaignSetupUI.UpdateSubList(SubmarineInfo.SavedSubmarines);
607  break;
608  case Tab.LoadGame:
609  campaignSetupUI.UpdateLoadMenu();
610  break;
611  case Tab.Settings:
612  SettingsMenu.Create(menuTabs[Tab.Settings].RectTransform);
