20  private static void _hookLuaCsPatch(MethodBase __originalMethod, object[] __args, object __instance, out object result, HookMethodType hookType)
21  {
22  result = null;
23 
24  try
25  {
26  var funcAddr = ((long)__originalMethod.MethodHandle.GetFunctionPointer());
27  HashSet<(string, LuaCsCompatPatchFunc, ACsMod)> methodSet = null;
28  switch (hookType)
29  {
30  case HookMethodType.Before:
31  instance.compatHookPrefixMethods.TryGetValue(funcAddr, out methodSet);
32  break;
33  case HookMethodType.After:
34  instance.compatHookPostfixMethods.TryGetValue(funcAddr, out methodSet);
35  break;
36  default:
37  throw new ArgumentException($"Invalid {nameof(HookMethodType)} enum value.", nameof(hookType));
38  }
39 
40  if (methodSet != null)
41  {
42  var @params = __originalMethod.GetParameters();
43  var args = new Dictionary<string, object>();
44  for (int i = 0; i < @params.Length; i++)
45  {
46  args.Add(@params[i].Name, __args[i]);
47  }
48 
49  var outOfScope = new HashSet<(string, LuaCsCompatPatchFunc, ACsMod)>();
50  foreach (var tuple in methodSet)
51  {
52  if (tuple.Item3 != null && tuple.Item3.IsDisposed)
53  {
54  outOfScope.Add(tuple);
55  continue;
56  }
57 
58  var patchResult = tuple.Item2(__instance, args);
59  // lua patches can return null if an exception is caught
60  if (patchResult == null) continue;
61  if (__originalMethod is MethodInfo mi && mi.ReturnType != typeof(void))
62  {
63  if (patchResult is DynValue luaResult)
64  {
65  // XXX: unfortunately checking for IsNil
66  // instead of IsVoid makes it impossible to
67  // replace the return value with "null".
68  if (!luaResult.IsNil())
69  {
70  result = luaResult.ToObject(mi.ReturnType);
71  }
72  }
73  else // only C# mods can return CLR types
74  {
75  result = patchResult;
76  }
77  }
78  }
79  foreach (var tuple in outOfScope) { methodSet.Remove(tuple); }
80  }
81  }
82  catch (Exception ex)
83  {
84  GameMain.LuaCs.HandleException(ex, $"Error in {__originalMethod.Name}:", exceptionType: LuaCsSetup.ExceptionType.Both);
85  }
86  }
