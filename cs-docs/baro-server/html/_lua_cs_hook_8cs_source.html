123  public static void LoadLocalAndCast(this Emit il, Local value, Type targetType)
124  {
125  if (value == null) throw new ArgumentNullException(nameof(value));
126  if (targetType == null) throw new ArgumentNullException(nameof(targetType));
127  if (value.LocalType != typeof(object))
128  {
129  throw new ArgumentException($"Expected local type {typeof(object)}; got {value.LocalType}.", nameof(value));
130  }
131 
132  var guid = Guid.NewGuid().ToString("N");
133 
134  if (targetType.IsByRef)
135  {
136  targetType = targetType.GetElementType();
137  }
138 
139  // IL: var baseType = value.GetType();
140  var baseType = il.DeclareLocal(typeof(Type), $"cast_baseType_{guid}");
141  il.LoadLocal(value);
142  il.Call(typeof(object).GetMethod("GetType"));
143  il.StoreLocal(baseType);
144 
145  // IL: var implicitConversionMethod = SigilExtensions.HasImplicitConversion(baseType, <targetType>);
146  var implicitConversionMethod = il.DeclareLocal(typeof(MethodInfo), $"cast_implicitConversionMethod_{guid}");
147  il.LoadLocal(baseType);
148  il.LoadType(targetType);
149  il.Call(typeof(SigilExtensions).GetMethod(nameof(HasImplicitConversion), BindingFlags.NonPublic | BindingFlags.Static));
150  il.StoreLocal(implicitConversionMethod);
151 
152  // IL: <TargetType> castValue;
153  var castValue = il.DeclareLocal(targetType, $"cast_castValue_{guid}");
154 
155  // IL: if (implicitConversionMethod != null)
156  il.LoadLocal(implicitConversionMethod);
157  il.Branch((il) =>
158  {
159  // IL: var methodInvokeParams = new object[1];
160  var methodInvokeParams = il.DeclareLocal(typeof(object[]), $"cast_methodInvokeParams_{guid}");
161  il.LoadConstant(1);
162  il.NewArray(typeof(object));
163  il.StoreLocal(methodInvokeParams);
164 
165  // IL: methodInvokeParams[0] = value;
166  il.LoadLocal(methodInvokeParams);
167  il.LoadConstant(0);
168  il.LoadLocal(value);
169  il.StoreElement<object>();
170 
171  // IL: castValue = (<TargetType>)implicitConversionMethod.Invoke(null, methodInvokeParams);
172  il.LoadLocal(implicitConversionMethod);
173  il.LoadNull(); // first parameter is null because implicit cast operators are static
174  il.LoadLocal(methodInvokeParams);
175  il.Call(typeof(MethodInfo).GetMethod("Invoke", new[] { typeof(object), typeof(object[]) }));
176  if (targetType.IsValueType)
177  {
178  il.UnboxAny(targetType);
179  }
180  else
181  {
182  il.CastClass(targetType);
183  }
184  il.StoreLocal(castValue);
185  },
186  (il) =>
187  {
188  // IL: castValue = (<TargetType>)value;
189  il.LoadLocal(value);
190  if (targetType.IsValueType)
191  {
192  il.UnboxAny(targetType);
193  }
194  else
195  {
196  il.CastClass(targetType);
197  }
198  il.StoreLocal(castValue);
199  });
200 
201  il.LoadLocal(castValue);
202  }

309  public static void ForEachEnumerator<T>(this Emit il, Local enumerator, Action<Emit, Local, Sigil.Label> action)
310  {
311  if (enumerator == null) throw new ArgumentNullException(nameof(enumerator));
312  if (action == null) throw new ArgumentNullException(nameof(action));
313  if (!typeof(IEnumerator<T>).IsAssignableFrom(enumerator.LocalType))
314  {
315  throw new ArgumentException($"Expected local type {typeof(IEnumerator<T>)}; got {enumerator.LocalType}.", nameof(enumerator));
316  }
317 
318  var guid = Guid.NewGuid().ToString("N");
319  var labelLoopStart = il.DefineLabel($"forEach_loopStart_{guid}");
320  var labelMoveNext = il.DefineLabel($"forEach_moveNext_{guid}");
321  var labelLeave = il.DefineLabel($"forEach_leave_{guid}");
322 
323  il.BeginExceptionBlock(out var exceptionBlock);
324  il.Branch(labelMoveNext); // MoveNext() needs to be called at least once before iterating
325  il.MarkLabel(labelLoopStart);
326 
327  // IL: var current = enumerator.Current;
328  var current = il.DeclareLocal<T>($"forEachEnumerator_current_{guid}");
329  il.LoadLocal(enumerator);
330  il.CallVirtual(enumerator.LocalType.GetProperty("Current").GetGetMethod());
331  il.StoreLocal(current);
332 
333  action(il, current, labelLeave);
334 
335  il.MarkLabel(labelMoveNext);
336  il.LoadLocal(enumerator);
337  il.CallVirtual(typeof(IEnumerator).GetMethod("MoveNext"));
338  il.BranchIfTrue(labelLoopStart); // loop if MoveNext() returns true
339 
340  // IL: finally { enumerator.Dispose(); }
341  il.BeginFinallyBlock(exceptionBlock, out var finallyBlock);
342  il.LoadLocal(enumerator);
343  il.CallVirtual(typeof(IDisposable).GetMethod("Dispose"));
344  il.EndFinallyBlock(finallyBlock);
345 
346  il.EndExceptionBlock(exceptionBlock);
347 
348  il.MarkLabel(labelLeave);
349  }

738  private Stopwatch performanceMeasurement = new Stopwatch();
739 
740  [MoonSharpHidden]
741  public T Call<T>(string name, params object[] args)
742  {
743  if (name == null) throw new ArgumentNullException(name);
744  if (args == null) args = new object[0];
745 
746  name = NormalizeIdentifier(name);
747  if (!hookFunctions.ContainsKey(name)) return default;
748 
749  T lastResult = default;
750 
751  var hooksToRemove = new List<string>();
752  foreach ((var key, var tuple) in hookFunctions[name])
753  {
754  if (tuple.Item2 != null && tuple.Item2.IsDisposed)
755  {
756  hooksToRemove.Add(key);
757  continue;
758  }
759 
760  try
761  {
762  if (LuaCs.PerformanceCounter.EnablePerformanceCounter)
763  {
764  performanceMeasurement.Start();
765  }
766 
767  var result = tuple.Item1.func(args);
768  // TODO(BREAKING): change this to !result.IsVoid()
769  if (result != null && !result.IsNil())
770  {
771  lastResult = result.ToObject<T>();
772  }
773 
774  if (LuaCs.PerformanceCounter.EnablePerformanceCounter)
775  {
776  performanceMeasurement.Stop();
777  LuaCs.PerformanceCounter.SetHookElapsedTicks(name, key, performanceMeasurement.ElapsedTicks);
778  performanceMeasurement.Reset();
779  }
780  }
781  catch (Exception e)
782  {
783  StringBuilder argsSb = new StringBuilder();
784  foreach (var arg in args) argsSb.Append(arg + " ");
785  GameMain.LuaCs.HandleException(e, $"Error in Hook '{name}'->'{key}', with args '{argsSb}':\n{e}", ExceptionType.Both);
786  }
787  }
788  foreach (var key in hooksToRemove)
789  {
790  hookFunctions[name].Remove(key);
791  }
792
